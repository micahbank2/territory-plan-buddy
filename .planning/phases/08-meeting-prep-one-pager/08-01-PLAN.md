```ts
export const SECTIONS = [
  "Context",
  "Recent History",
  "Contacts",
  "Open Tasks",
  "Talking Points",
  "Suggested Ask",
] as const;

export interface MeetingBrief {
  context: string;          // 2-3 sentences
  recentHistory: string;    // bulleted list (markdown source)
  contacts: string;         // bulleted list (markdown source)
  openTasks: string;        // bulleted list (markdown source)
  talkingPoints: string;    // 3-5 bullets, Yext-anchored (markdown source)

}
```

**Parser algorithm (Task 2):**

```ts
export function parseMeetingBrief(md: string): MeetingBrief {
  const lines = md.split("\n");
  const sections: Record<string, string[]> = {};
  let current = "";
  for (const line of lines) {
    const m = line.match(/^##\s+(.+?)\s*$/);
    if (m && (SECTIONS as readonly string[]).includes(m[1])) {
      current = m[1];
      sections[current] = [];
    } else if (current) {
      sections[current].push(line);
    }
  }
  const get = (k: SectionName) => (sections[k] ?? []).join("\n").trim();
  return {
    context: get("Context"),
    recentHistory: get("Recent History"),
    contacts: get("Contacts"),
    openTasks: get("Open Tasks"),
    talkingPoints: get("Talking Points"),
    suggestedAsk: get("Suggested Ask"),
    raw: md,
  };
}
```

    ```ts
    export function parseMeetingBrief(md: string): MeetingBrief {
      const lines = md.split("\n");
      const sections: Record<string, string[]> = {};
      let current = "";
      for (const line of lines) {
        const m = line.match(/^##\s+(.+?)\s*$/);
        if (m && (SECTIONS as readonly string[]).includes(m[1])) {
          current = m[1];
          sections[current] = [];
        } else if (current) {
          sections[current].push(line);
        }
      }
      const get = (k: SectionName) => (sections[k] ?? []).join("\n").trim();
      return {
        context: get("Context"),
        recentHistory: get("Recent History"),
        contacts: get("Contacts"),
        openTasks: get("Open Tasks"),
        talkingPoints: get("Talking Points"),
        suggestedAsk: get("Suggested Ask"),
        raw: md,
      };
    }
    ```
