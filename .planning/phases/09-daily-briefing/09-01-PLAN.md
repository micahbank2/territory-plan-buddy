export interface BriefingItem {

  name: string;
  reason: string;        // pre-formatted display string e.g. "Hot ┬Ę 45d since touch ┬Ę score 72"

  daysStale: number | null;  // null when never contacted

export interface OverdueTaskItem {
  prospectId: string;
  prospectName: string;
  taskId: string;
  text: string;
  dueDate: string;       // YYYY-MM-DD
  daysOverdue: number;   // positive integer
}

export interface OppMovementItem {
  oppId: string;
  name: string;
  stage: string;
  potentialValue: number;
  daysSinceCreated: number;
  prospectId: string | null;
}

export interface Briefing {
  generatedAt: string;            // today.toISOString()
  todayLabel: string;             // "Saturday, April 25, 2026" (en-US)
  hero: {
    activeProspects: number;      // count where status !== "Churned" && status !== "Closed Lost Prospect"
    hotCount: number;             // count where priority === "Hot" (active only)
    weightedPipeline: number;     // forecastPipeline(opportunities, 0).weighted
    overdueTaskCount: number;     // total count BEFORE the cap-10 truncation

}

export function getBriefing(

  opportunities: Opportunity[],

**Engine algorithm (Task 2 implements; Task 1 stubs):**

```ts
const daysBetween = (iso: string, today: Date): number =>
  Math.floor((today.getTime() - new Date(iso).getTime()) / 86400000);

const todayStr = today.toISOString().split("T")[0]; // YYYY-MM-DD

// 1. Active filter (used by all hero counts and section filters)
const active = prospects.filter(p => p.status !== "Churned" && p.status !== "Closed Lost Prospect");

// 2. Today's Plan ŌĆö Hot prospects either never contacted OR > 14 days stale
const todayPlan: BriefingItem[] = active
  .filter(p => p.priority === "Hot")
  .filter(p => p.lastTouched == null || daysBetween(p.lastTouched, today) >= 14)
  .map(p => {
    const score = scoreProspect(p);
    const daysStale = p.lastTouched == null ? null : daysBetween(p.lastTouched, today);
    const reason = daysStale == null
      ? `Hot ┬Ę never contacted ┬Ę score ${score}`
      : `Hot ┬Ę ${daysStale}d since touch ┬Ę score ${score}`;
    return { prospectId: String(p.id), name: p.name, reason, score, daysStale };
  })
  .sort((a, b) => b.score - a.score)
  .slice(0, 5);

// 3. Overdue Tasks ŌĆö flatten and sort
const overdueAll: OverdueTaskItem[] = [];
for (const p of active) {
  for (const t of (p.tasks || [])) {
    if (!t.dueDate || t.dueDate >= todayStr) continue;
    overdueAll.push({
      prospectId: String(p.id),
      prospectName: p.name,
      taskId: t.id,
      text: t.text,
      dueDate: t.dueDate,
      daysOverdue: daysBetween(t.dueDate, today),
    });
  }
}
overdueAll.sort((a, b) => b.daysOverdue - a.daysOverdue);
const overdueTasks = overdueAll.slice(0, 10);

// 4. Going Stale ŌĆö Hot/Warm + lastTouched non-null + 30d+ + score>=40
const goingStale: BriefingItem[] = active
  .filter(p => p.priority === "Hot" || p.priority === "Warm")
  .filter(p => p.lastTouched != null && daysBetween(p.lastTouched, today) >= 30)
  .map(p => {
    const score = scoreProspect(p);
    const daysStale = daysBetween(p.lastTouched!, today);
    return {
      prospectId: String(p.id),
      name: p.name,
      reason: `${p.priority} ┬Ę ${daysStale}d since touch ┬Ę score ${score}`,
      score,
      daysStale,
    };
  })
  .filter(item => item.score >= 40)
  .sort((a, b) => b.score - a.score)
  .slice(0, 10);

// 5. New Pipeline ŌĆö opps with created_at in last 7d
const newPipeline: OppMovementItem[] = opportunities
  .filter(o => {
    if (!o.created_at) return false;
    const days = daysBetween(o.created_at, today);
    return days >= 0 && days <= 7;
  })
  .map(o => ({
    oppId: o.id,
    name: o.name,
    stage: o.stage,
    potentialValue: o.potential_value || 0,
    daysSinceCreated: daysBetween(o.created_at, today),
    prospectId: o.prospect_id,
  }))
  .sort((a, b) => a.daysSinceCreated - b.daysSinceCreated);

// 6. Hero ŌĆö counts + reuse forecastPipeline for weighted
const forecast = forecastPipeline(opportunities, 0);
const hero = {
  activeProspects: active.length,
  hotCount: active.filter(p => p.priority === "Hot").length,
  weightedPipeline: forecast.weighted,
  overdueTaskCount: overdueAll.length, // BEFORE cap
};

// 7. Inbox zero
const inboxZero = todayPlan.length === 0 && overdueTasks.length === 0 && goingStale.length === 0;

return {
  generatedAt: today.toISOString(),
  todayLabel: today.toLocaleDateString("en-US", { weekday: "long", month: "long", day: "numeric", year: "numeric" }),
  hero,
  todayPlan,
  overdueTasks,
  goingStale,
  newPipeline,
  inboxZero,
};
```
