```typescript
// remove() — soft delete
const remove = useCallback(async (id: string) => {
  if (!user) return;
  const { error } = await supabase
    .from("prospects")
    .update({ deleted_at: new Date().toISOString() })
    .eq("id", id);
  if (error) { toast.error("Failed to archive prospect"); return; }
  setData(prev => prev.filter(p => p.id !== id));
}, [user]);

// restore() — set deleted_at to null
const restore = useCallback(async (id: string) => {
  if (!user) return;
  const { error } = await supabase
    .from("prospects")
    .update({ deleted_at: null })
    .eq("id", id);
  if (error) { toast.error("Failed to restore prospect"); return; }
  setArchivedData(prev => prev.filter(p => p.id !== id));
  // Trigger reload of main list to include restored prospect
  await loadData();
}, [user, loadData]);

// permanentDelete() — hard delete (from archive only)
const permanentDelete = useCallback(async (id: string) => {
  if (!user) return;
  const { error } = await supabase.from("prospects").delete().eq("id", id);
  if (error) { toast.error("Failed to permanently delete prospect"); return; }
  setArchivedData(prev => prev.filter(p => p.id !== id));
}, [user]);
```

<!-- CRITICAL RLS rule (from RESEARCH.md Pitfall 2 and D-12) -->
The `deleted_at IS NULL` filter MUST be in the application query (.is("deleted_at", null)), NOT in the Supabase RLS SELECT policy. Adding it to RLS blocks the soft-delete UPDATE because Supabase re-evaluates post-update rows against WITH CHECK. Apply in code queries only.
