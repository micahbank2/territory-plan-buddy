TerritoryDialogGroup.tsx:
```ts
import type { User } from "@supabase/supabase-js";
import type { Prospect, InteractionLog } from "@/data/prospects";
import type { Territory, TerritoryMember } from "@/hooks/useTerritories";
import type { Signal } from "@/hooks/useSignals";

export interface TerritoryDialogGroupHandle {
  openAdd: () => void;
  openAddContact: () => void;
  openUpload: () => void;
  openPasteImport: () => void;
  openEnrich: () => void;
  openExport: () => void;
  openShare: () => void;
  openNewTerritory: () => void;
  openReset: () => void;
  openDeleteConfirm: (prospectId: string) => void;
}

export interface TerritoryDialogGroupProps {
  // Auth + territory context
  user: User | null;
  territoryId: string | null;
  myRole: "owner" | "editor" | "viewer";
  territories: Territory[];
  activeTerrObj: Territory | null;
  members: TerritoryMember[];

  // Full data set (Add/Import/Export need existing names for dedupe; Export takes filtered)
  data: Prospect[];
  filtered: Prospect[];       // ExportDialog receives the filtered set
  signals: Signal[];          // ContactPicker and some dialogs read signals

  // Prospect mutations
  add: (partial: Partial<Prospect>) => Promise<string>;
  bulkAdd: (partials: Partial<Prospect>[]) => Promise<void>;
  bulkMerge: (updates: Partial<Prospect>[]) => Promise<void>;
  update: (id: string, changes: Partial<Prospect>) => Promise<void>;
  remove: (id: string) => Promise<void>;
  addNote: (prospectId: string, text: string) => Promise<void>;
  addContact: (prospectId: string, contact: any) => Promise<void>;
  addInteraction: (prospectId: string, i: Omit<InteractionLog, "id">) => Promise<void>;

  // Territory mutations
  createTerritory: (name: string) => Promise<void>;
  renameTerritory: (id: string, name: string) => Promise<void>;
  inviteMember: (email: string, role: "editor" | "viewer") => Promise<void>;
  removeMember: (userId: string) => Promise<void>;
  updateMemberRole: (userId: string, role: "editor" | "viewer") => Promise<void>;

  // Reset flow (owner-only)
  reset: () => Promise<void>;

  // Shared styling passthroughs used by inline inputs in the AddProspect/NewTerritory dialogs
  inputClass: string;
  selectClass: string;

  // Coordinator-owned state the delete flow needs to clear
  onProspectDeleted: (id: string) => void;  // called after remove() so coordinator can setSheetProspectId(null)
}

export const TerritoryDialogGroup = forwardRef<TerritoryDialogGroupHandle, TerritoryDialogGroupProps>(...)
```
